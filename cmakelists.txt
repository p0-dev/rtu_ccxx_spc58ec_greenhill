# @location     | PJT_ROOT_DIR_PATH
# @brief        | CMake main scrip file
# ****************************************************************************************
# @desc         | Include
#               |   + Target list
#               |   + CMake general settings
#               |   + Sub-modules
# ****************************************************************************************


# ========================================================================================
# Target List
# ========================================================================================
# [Name]            | [Location]    | [Description]
# ----------------------------------------------------------------------------------------
# lib_main          | src/main      | main execution of the project
# ----------------------------------------------------------------------------------------

# ========================================================================================
# Pre Processing
# ========================================================================================
cmake_minimum_required  (VERSION 3.29)
project                 (${PJT_NAME})

# ========================================================================================
# General Configuration
# ========================================================================================
set (CMAKE_EXPORT_COMPILE_COMMANDS 1)

# ========================================================================================
# Module
# ========================================================================================
add_subdirectory (${CMAKE_SOURCE_DIR}/src/main)

# ========================================================================================
# Target
# ========================================================================================
# * Note : no source file project target
add_executable (${PROJECT_NAME})

# ========================================================================================
# Properties
# ========================================================================================
set_target_properties (
  ${PROJECT_NAME}
  PROPERTIES
  SUFFIX ".elf"
)

# ========================================================================================
# Final Target Linking Library
# ========================================================================================
target_link_libraries (
  ${PROJECT_NAME}
  PUBLIC
    lib_main
)

# ========================================================================================
# Final Target Installation
# ========================================================================================
install (
  TARGETS ${PROJECT_NAME} 
  DESTINATION .
)
